## 제출한 답
* 빠르게 풀려다가 코드가 너무 더러워졌다

```py
def digits(N): #
    result = set()
    while N > 0: 
        result.add(N % 10)
        N //= 10
    return result

T = int(input())

for t in range(1, T + 1):
    data = [0 for _ in range(10)]
    N = int(input())
    i = 0
    while True:
        i += 1
        n = N * i
        for num in digits(n):
            data[num] = 1
        if data.count(0):
            continue
        else:
            break
        
    print(f'#{t} {n}')
```

* 함수 만들필요x
* while 탈출 더 쉽게
* set 활용
* 배수 설정 더 쉽게

t = int(input())
dap = []
 
for i in range(1, t+1):
 
    n = int(input())
 
    ch = [False] * 10
 
    num = 1
    number = 0
    while False in ch:
        tmp = n * num
        number = tmp
        while tmp >= 1:

            ch[b - 1] = True
